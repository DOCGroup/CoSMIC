00070 /* This class is a smart pointer implementation over DOMNode interface and
00071 ** classes derived from it. It takes care of reference counting automatically.
00072 ** Reference counting is optional so use of this class is experimental.
00073 */
00074 template <class T> class StDOMNode {
00075     T* m_node;
00076 
00077     static inline void INCREFCOUNT(T *x) { if (x != (T*)0) x->incRefCount(); }
00078     static inline void DECREFCOUNT(T *x) { if (x != (T*)0) x->decRefCount(); }
00079 
00080 public:
00081     inline StDOMNode(T* node = (T*)0) : m_node(node) { INCREFCOUNT(m_node); }
00082     inline StDOMNode(const StDOMNode& stNode) : m_node(stNode.m_node) { INCREFCOUNT(m_node); }
00083     inline ~StDOMNode() { DECREFCOUNT(m_node); }
00084 
00085     inline T* operator= (T *node)
00086     {
00087         if (m_node != node) {
00088             DECREFCOUNT(m_node);
00089             m_node = node;
00090             INCREFCOUNT(m_node);
00091         }
00092         return (m_node);
00093     }
00094 
00095     inline bool operator!= (T* node) const { return (m_node != node); }
00096     inline bool operator== (T* node) const { return (m_node == node); }
00097 
00098     inline T& operator* () { return (*m_node); }
00099     inline const T& operator* () const { return (*m_node); }
00100     inline T* operator-> () const { return (m_node); }
00101     inline operator T*() const { return (m_node); }
00102     inline void ClearNode() { operator=((T*)(0)); }
00103 };
